### 模块1：generate-feed.js（发现热点）

#### 参考 follow-builders 的设计
```javascript
// follow-builders/scripts/generate-feed.js 的核心逻辑：
// 1. 读取配置文件（default-sources.json）
// 2. 按源类型执行抓取（X、Podcasts、Blogs）
// 3. 时间过滤（lookback hours）
// 4. 去重（state-feed.json）
// 5. 输出JSON文件
```

#### 我们的实现
```javascript
#!/usr/bin/env node

import { readFile, writeFile } from 'fs/promises';
import { existsSync } from 'fs';

// 常量
const LOOKBACK_HOURS = 24;
const STATE_PATH = './state/state-feed.json';
const OUTPUT_PATH = './data/feeds/feed-{date}.json';

// 主函数
async function main() {
  // 1. 加载配置
  const sources = await loadSources('./config/car-sources.json');
  const state = await loadState();
  
  // 2. 抓取各类信息源
  const results = [];
  
  // 2.1 懂车帝创作者
  for (const creator of sources.懂车帝创作者) {
    if (!creator.enabled) continue;
    const items = await fetchDongchediCreator(creator);
    results.push(...items);
  }
  
  // 2.2 行业Builder微博
  for (const builder of sources.行业Builder_微博) {
    if (!builder.enabled) continue;
    const items = await fetchWeiboUser(builder);
    results.push(...items);
  }
  
  // 2.3 新车发布网页
  for (const site of sources.新车发布_网页) {
    if (!site.enabled) continue;
    const items = await scrapeNewCarSite(site);
    results.push(...items);
  }
  
  // 2.4 Tavily搜索
  for (const query of sources.Tavily搜索) {
    if (!query.enabled) continue;
    const items = await searchTavily(query);
    results.push(...items);
  }
  
  // 3. 时间过滤
  const cutoff = Date.now() - LOOKBACK_HOURS * 60 * 60 * 1000;
  const filtered = results.filter(item => {
    const publishTime = new Date(item.发布时间).getTime();
    return publishTime >= cutoff;
  });
  
  // 4. 去重（基于state）
  const deduplicated = filtered.filter(item => {
    if (state.seenArticles[item.id]) return false;
    state.seenArticles[item.id] = Date.now();
    return true;
  });
  
  // 5. 提取关键词
  for (const item of deduplicated) {
    item.关键词 = await extractKeywords(item.标题, item.内容摘要);
  }
  
  // 6. 输出
  const output = {
    生成时间: new Date().toISOString(),
    时间窗口: '24h',
    总文章数: deduplicated.length,
    热点列表: deduplicated
  };
  
  await writeFile(OUTPUT_PATH, JSON.stringify(output, null, 2));
  await saveState(state);
}

// 抓取函数示例

  // 返回格式：[{ id, 标题, 链接, 来源, 发布时间, 内容摘要, 热度指数, 媒体类型 }]
}

```

#### 关键技术点
1. **状态管理**：参考 follow-builders 的 `state-feed.json`，记录已见过的文章ID
2. **时间过滤**：只保留24小时内的内容

### 模块3：rank-topics.js（筛选与排序）

#### 实现逻辑
```javascript
#!/usr/bin/env node

import { readFile, writeFile } from 'fs/promises';

async function main() {
  // 1. 读取模块2输出
  const clustered = JSON.parse(await readFile('./data/clustered/clustered-{date}.json'));
  
  // 2. 读取评分Prompt
  const prompts = JSON.parse(await readFile('./config/ranking-prompts.json'));
  
  // 3. Gemini初评
  const geminiScores = [];
  for (const topic of clustered.话题分组) {
    const score = await callGemini(prompts.gemini_prompt, topic);
    geminiScores.push(score);
  }
  
  // 4. Claude复评
  const claudeScores = [];
  for (let i = 0; i < clustered.话题分组.length; i++) {
    const topic = clustered.话题分组[i];
    const geminiResult = geminiScores[i];
    const score = await callClaude(prompts.claude_prompt, topic, geminiResult);
    claudeScores.push(score);
  }
  
  // 5. Claude最终决策
  const finalDecision = await callClaude(
    prompts.claude_final_decision_prompt,
    { topics: clustered.话题分组, geminiScores, claudeScores }
  );
  
  // 6. 输出
  await writeFile('./data/ranked/ranked-{date}.json', JSON.stringify(finalDecision, null, 2));
}

main();
```

### 模块4：write-articles.js（写文章）

#### 实现逻辑
```javascript
#!/usr/bin/env node

import { readFile, writeFile } from 'fs/promises';

async function main() {
  // 1. 读取模块3输出
  const ranked = JSON.parse(await readFile('./data/ranked/ranked-{date}.json'));
  
  // 2. 读取写作Prompt
  const prompts = JSON.parse(await readFile('./config/writing-prompts.json'));
  
  // 3. 对每个Top 3话题写文章
  for (const topic of ranked.推荐列表) {
    // 3.1 Tavily深度搜索
    const additionalMaterials = await searchTavilyDeep(topic);
    const allMaterials = [...topic.相关文章, ...additionalMaterials];
    
    // 3.2 访问链接，提取完整内容
    const fullContents = await fetchAllContents(allMaterials);
    
    // 3.3 Gemini生成初稿
    const draft = await callGemini(prompts.gemini_draft_prompt, {
      topic: topic.话题,
      materials: fullContents
    });
    
    // 3.4 Claude优化润色
    const polished = await callClaude(prompts.claude_polish_prompt, draft);
    
    // 3.5 生成多平台版本
    const platforms = {};
    for (const [platform, config] of Object.entries(prompts.platform_prompts)) {
      platforms[platform] = await callClaude(config.prompt, polished);
    }
    
    // 3.6 输出
    const output = {
      生成时间: new Date().toISOString(),
      话题ID: topic.话题ID,
      话题: topic.话题,
      素材统计: {
        模块3已有: topic.相关文章.length,
        Tavily搜索: additionalMaterials.length,
        总计: allMaterials.length
      },
      通用版本: polished,
      ...platforms
    };
    
    await writeFile(`./data/articles/article-${topic.话题ID}-{date}.json`, JSON.stringify(output, null, 2));
  }
}

main();
```

// follow-builders/scripts/deliver.js 的核心逻辑：

```javascript
#!/usr/bin/env node

import { readFile } from 'fs/promises';
import TelegramBot from 'node-telegram-bot-api';

async function main() {
  // 1. 读取配置
  const config = JSON.parse(await readFile('./config/telegram-config.json'));
  const bot = new TelegramBot(config.bot_token, { polling: false });
  
  // 2. 读取文章
  const ranked = JSON.parse(await readFile('./data/ranked/ranked-{date}.json'));
  
  // 3. 发送Top 3摘要
  for (const topic of ranked.推荐列表) {
    const message = formatSummary(topic, config.message_templates.summary);
    const buttons = createButtons(topic, config.buttons);
    
    await bot.sendMessage(config.chat_id, message, {
      parse_mode: 'Markdown',
      reply_markup: { inline_keyboard: buttons }
    });
  }
  
  // 4. 监听用户点击
  bot.on('callback_query', async (query) => {
    const data = query.data;
    
    if (data.startsWith('view_')) {
      // 发送完整文章
      const [action, platform, topicId] = data.split('_');
      const article = await loadArticle(topicId);
      await bot.sendMessage(query.message.chat.id, article[platform].正文);
    }
    
    if (data.startsWith('download_')) {
      // 发送PDF文件
      const topicId = data.replace('download_', '');
      const pdf = await generatePDF(topicId);
      await bot.sendDocument(query.message.chat.id, pdf);
    }
  });
}

main();
```

### 缓存机制
```javascript
// 缓存Tavily搜索结果（1小时）
const cache = new Map();
async function searchTavilyWithCache(query) {
  const key = `tavily:${query}`;
  if (cache.has(key)) {
    const { data, timestamp } = cache.get(key);
    if (Date.now() - timestamp < 3600000) return data;
  }
  const data = await searchTavily(query);
  cache.set(key, { data, timestamp: Date.now() });
  return data;
}
```

### 错误重试
```javascript
async function retryWithBackoff(fn, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (err) {
      if (i === maxRetries - 1) throw err;
      await sleep(1000 * Math.pow(2, i)); // 指数退避
    }
  }
}
```
